function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); // Cálculo de promedio var p1 = Number(data.p1) || 0; var p2 = Number(data.p2) || 0; var p3 = Number(data.p3) || 0; var p4 = Number(data.p4) || 0; var p5 = Number(data.p5) || 0; var p6 = Number(data.p6) || 0; var promedio = ((p1 + p2 + p3 + p4 + p5 + p6) / 6).toFixed(1); // Guardar en el Excel (Orden de columnas: Fecha, Prov, Nombres, Ruts, Equipo, P1..P6, Prom, Coment, Firma) sheet.appendRow([ data.fecha, data.proveedor, data.nombres, data.ruts, data.equipo, p1, p2, p3, p4, p5, p6, promedio, data.comentarios, data.firma ]); return ContentService.createTextOutput("OK").setMimeType(ContentService.MimeType.TEXT); } catch (err) { return ContentService.createTextOutput("Error: " + err.toString()); } }
